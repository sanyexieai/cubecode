1//! 多后端注册与带重试的 generate / generate_stream；可选 [`crate::pipeline::Pipeline`]。
2//!
3//! 挂 pipeline 时的调用顺序与语义见 [`crate::pipeline`] 模块文档（**单次** provider + 前后 `PipelineStage` 链）。
4
5use std::collections::BTreeMap;
6use std::thread;
7
8use crate::core::error::LlmError;
9use crate::core::provider::LlmProvider;
10use crate::core::types::{GenerateRequest, GenerateResponse, ProviderInfo, StreamChunk};
11use crate::pipeline::{Pipeline, PipelineContext};
12use crate::runtime::retry::LlmRetryPolicy;
13
14pub struct ProviderRegistry {
15    providers: BTreeMap<String, Box<dyn LlmProvider>>,
16    retry_policy: LlmRetryPolicy,
17    pipeline: Option<Pipeline>,
18}
19
20impl Default for ProviderRegistry {
21    fn default() -> Self {
22        Self {
23            providers: BTreeMap::new(),
24            retry_policy: LlmRetryPolicy::from_env(),
25            pipeline: None,
26        }
27    }
28}
29
30impl ProviderRegistry {
31    pub fn new() -> Self {
32        Self::default()
33    }
34
35    /// 设置贯穿 generate / generate_stream 的流水线；`None` 或空 [`Pipeline`] 均不改变原有逻辑路径。
36    pub fn set_pipeline(&mut self, pipeline: Option<Pipeline>) {
37        self.pipeline = pipeline.filter(|p| !p.is_empty());
38    }
39
40    pub fn pipeline(&self) -> Option<&Pipeline> {
41        self.pipeline.as_ref()
42    }
43
44    pub fn set_retry_policy(&mut self, policy: LlmRetryPolicy) {
45        self.retry_policy = policy;
46    }
47
48    pub fn retry_policy(&self) -> LlmRetryPolicy {
49        self.retry_policy
50    }
51
52    pub fn register<P>(&mut self, provider: P)
53    where
54        P: LlmProvider + 'static,
55    {
56        let id = provider.info().id;
57        self.providers.insert(id, Box::new(provider));
58    }
59
60    pub fn provider(&self, id: &str) -> Option<&dyn LlmProvider> {
61        self.providers.get(id).map(|provider| provider.as_ref())
62    }
63
64    pub fn generate(&self, request: &GenerateRequest) -> Result<GenerateResponse, LlmError> {
65        if let Some(pipe) = self.pipeline.as_ref().filter(|p| !p.is_empty()) {
66            let mut ctx = PipelineContext::new(request.clone());
67            pipe.run_before(&mut ctx)?;
68            let response = self.generate_core(&ctx.request)?;
69            ctx.response = Some(response);
70            pipe.run_after_generate(&mut ctx)?;
71            return Ok(ctx.take_response());
72        }
73        self.generate_core(request)
74    }
75
76    pub fn generate_stream(
77        &self,
78        request: &GenerateRequest,
79        on_chunk: &mut dyn FnMut(StreamChunk) -> Result<(), LlmError>,
80    ) -> Result<GenerateResponse, LlmError> {
81        if let Some(pipe) = self.pipeline.as_ref().filter(|p| !p.is_empty()) {
82            let mut ctx = PipelineContext::new(request.clone());
83            pipe.run_before(&mut ctx)?;
84            let response = self.generate_stream_core(&ctx.request, on_chunk)?;
85            ctx.response = Some(response);
86            pipe.run_after_stream(&mut ctx)?;
87            return Ok(ctx.take_response());
88        }
89        self.generate_stream_core(request, on_chunk)
90    }
91
92    fn generate_core(&self, request: &GenerateRequest) -> Result<GenerateResponse, LlmError> {
93        let provider = self
94            .provider(&request.model.provider)
95            .ok_or_else(|| LlmError::ProviderNotFound(request.model.provider.clone()))?;
96        let policy = self.retry_policy;
97        let mut attempt = 0usize;
98        loop {
99            attempt += 1;
100            match provider.generate(request) {
101                Ok(response) => return Ok(response),
102                Err(error) if policy.should_retry(attempt, &error) => {
103                    let delay = policy.backoff_for_attempt(attempt);
104                    policy.log_retry(attempt, &error, delay);
105                    thread::sleep(delay);
106                }
107                Err(error) => return Err(error),
108            }
109        }
110    }
111
112    fn generate_stream_core(
113        &self,
114        request: &GenerateRequest,
115        on_chunk: &mut dyn FnMut(StreamChunk) -> Result<(), LlmError>,
116    ) -> Result<GenerateResponse, LlmError> {
117        let provider = self
118            .provider(&request.model.provider)
119            .ok_or_else(|| LlmError::ProviderNotFound(request.model.provider.clone()))?;
120        let policy = self.retry_policy;
121        let mut attempt = 0usize;
122        loop {
123            attempt += 1;
124            let mut saw_chunk = false;
125            let mut wrapped = |chunk: StreamChunk| -> Result<(), LlmError> {
126                saw_chunk = true;
127                on_chunk(chunk)
128            };
129            match provider.generate_stream(request, &mut wrapped) {
130                Ok(response) => return Ok(response),
131                Err(error) if !saw_chunk && policy.should_retry(attempt, &error) => {
132                    let delay = policy.backoff_for_attempt(attempt);
133                    policy.log_retry(attempt, &error, delay);
134                    thread::sleep(delay);
135                }
136                Err(error) => return Err(error),
137            }
138        }
139    }
140
141    pub fn provider_infos(&self) -> Vec<ProviderInfo> {
142        self.providers
143            .values()
144            .map(|provider| provider.info())
145            .collect()
146    }
147}
