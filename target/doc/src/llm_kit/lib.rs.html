1//! 通用可插拔 LLM 核心：多种线路协议（Chat Completions、Anthropic Messages）、流式、重试与注册表。
2//!
3//! 厂商与协议见 [`providers`] 模块文档。
4//!
5//! ## 源码目录（与 `src/` 下文件夹一致）
6//! - `core/` — 错误、类型、[`LlmProvider`] 契约
7//! - `runtime/` — 环境变量与重试策略
8//! - `registry/` — 多后端注册与带重试的调用（可选 pipeline）
9//! - `sanitize/` — 助手文本清洗
10//! - `providers/` — 厂商预设与线路协议实现
11//! - `pipeline/` — 可选请求→响应流水线（无节点时不改行为）
12//!
13//! 环境变量优先读取 `LLM_*`，并兼容 Honeycomb 风格的 `HC_LLM_*`。
14//!
15//! 库本身只读进程环境，不读取 `.env` 文件；**`apps/llm-cli` 提供的 `llm-kit` 命令** 会在启动时向上查找并加载 `.env`。

50pub fn default_registry_from_env() -> ProviderRegistry {
51    let mut registry = ProviderRegistry::new();
52    let provider_id = default_provider_from_env();
53    let api_key = provider_api_key_from_env(&provider_id);
54    let base_url = provider_base_url_from_env(&provider_id);
55
56    let Some(api_key) = api_key else {
57        return registry;
58    };
59
60    let display_name = provider_preset(&provider_id)
61        .map(|preset| preset.display_name.to_owned())
62        .unwrap_or_else(|| format!("{provider_id} compatible"));
63
64    let wire = provider_preset(&provider_id)
65        .map(|preset| preset.wire)
66        .unwrap_or(WireProtocol::ChatCompletions);
67
68    match wire {
69        WireProtocol::AnthropicMessages => {
70            if let Ok(provider) =
71                AnthropicMessagesProvider::new(provider_id.clone(), display_name, base_url, api_key)
72            {
73                registry.register(provider);
74            }
75        }
76        WireProtocol::ChatCompletions => {
77            if let Ok(provider) =
78                OpenAiCompatibleProvider::new(provider_id.clone(), display_name, base_url, api_key)
79            {
80                registry.register(provider);
81            }
82        }
83    }
84
85    registry
86}

88pub fn is_timeout_error(error: &LlmError) -> bool {
89    match error {
90        LlmError::ProviderFailure(message) => {
91            let lowered = message.to_ascii_lowercase();
92            lowered.contains("timed out") || lowered.contains("timeout")
93        }
94        LlmError::ProviderNotFound(_) | LlmError::InvalidRequest(_) => false,
95    }
96}
